import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int h = sc.nextInt();
        int r = sc.nextInt();
        int d = sc.nextInt();
        int t = sc.nextInt();
        int adv = sc.nextInt();

        int cost = (h * r) + d + t - adv;

        System.out.println("Final Project Cost = " + cost);
    }
}
